\begin{titlepage}

  %% centered single-spaced title page w/o page numbering
  \begin{center} \singlespacing \thispagestyle{empty}

    %%%% thesis title: 1.5 inches from the top of the page
    \ThesisTitle{\Title}

    \vspace{1in}                    % gap between the title and the
    % author: approx. 1 inch

    \ThesisAuthor{\Author}         % author name for the thesis

    \vspace{1.5in}                  % gap between the author and
    % statement: 1.5 inches

    %%%% for masters, change the arguments to: {thesis}{masters program name}
    \ifx\Degree\masters
      \ThesisStatement{thesis}{Masters Program Name}
    \else
      \ThesisStatement{dissertation}{Doctor of Philosophy}
    \fi
    %% gap between statement and location: approx. 0.5 inch
    \vspace{0.5in} \\

    \Location \\                      % prints Baltimore, Maryland as the location
    \ThesisDate{\Month}{\Year}        % thesis submission month and
    % year (single-spaced)

    %%%% optional copyright statement: approx. 2 inches from the
    % bottom of the page
    %% year and name are input argument for copyright statement
    \ThesisCopyright{\Year}{\Author}

  \end{center}

\end{titlepage}
